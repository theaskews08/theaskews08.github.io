document.addEventListener("DOMContentLoaded", function () { fetch('plu_codes.json') .then(response => response.json()) .then(data => { const df = data; const searchInput = document.getElementById("searchInput"); const outputLabel = document.getElementById("outputLabel"); const clearButton = document.getElementById("clearButton"); const exitButton = document.getElementById("exitButton"); searchInput.addEventListener("input", function () { const searchFor = searchInput.value.trim(); if (searchFor) { const results = df.filter( (row) => row["Name"].toLowerCase().includes(searchFor.toLowerCase()) || row["PLU_Code"].includes(searchFor) ); const resultsStr = results .map((row) => `${row["PLU_Code"]} - ${row["Name"]}`) .join("\n"); outputLabel.textContent = resultsStr; } else { outputLabel.textContent = ""; } }); clearButton.addEventListener("click", function () { searchInput.value = ""; outputLabel.textContent = ""; }); exitButton.addEventListener("click", function () { window.close(); }); }); });
